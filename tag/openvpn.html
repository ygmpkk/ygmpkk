"Python学习之路 - 处理Html"

Mon 14 January 2013
By Timothy Yeh

In config.

tags: openvpnvpnlinuxarchlinuxraspberry pi

运行环境： 硬件：Raspberry PI 系统：ArchLinux ARM

安装openvpn的包

# pacman -S openvpn

开始安装不上，原来是SD卡容量用完了

# df -h -d 1 /

找到一些不用，而且占用空间的数据，删除掉；

服务器端： 先到openvpn的目录去

# cd /usr/share/openvpn/easy-rsa

编辑vars文件

# vim ./vars

把一些基本的资料填上

# export KEY_COUNTRY="CN"
# export KEY_PROVINCE="SH"
# export KEY_CITY="Shanghai"
# export KEY_ORG="user"
# export KEY_EMAIL="user@mail.com"
# export KEY_CN=user
# export KEY_NAME=user
# export KEY_OU=user
# export PKCS11_MODULE_PATH=user
# export PKCS11_PIN=123456

# source ./vars
# ./clean-all
# ./build-dh
# ./pkitool --initca
# ./pkitool --server server
# ./pkitool client1
# ./pkitool --pass client2

编辑server的配置

# cp /usr/share/openvpn/example/server.conf /etc/openvpn/

# cp /usr/share/openvpn/easy-rsa/keys/ca.crt \
/usr/share/openvpn/easy-rsa/keys/server.crt \
/usr/share/openvpn/easy-rsa/keys/server.key \
/usr/share/openvpn/easy-rsa/keys/dh1024.pem \
/etc/openvpn/

# vim /etc/openvpn/server.conf

port 1194 #openvpn默认端口为1194
proto udp
dev tun 
ca ca.crt 
cert server.crt 
key server.key 
dh dh1024.pem 
server 172.16.0.0 255.255.0.0 
push "dhcp-option DNS 168.95.192.1"
push "dhcp-option DNS 168.95.192.2"
push "route 172.16.0.0 255.255.0.0"
ifconfig-pool-persist ipp.txt 
keepalive 10 120 
comp-lzo 
user nobody 
group nobody 
persist-key 
persist-tun 
status /var/log/openvpn-status.log
verb 3 
client-to-client
duplicate-cn
log /var/log/openvpn.log
log-append /var/log/openvpn.log

客户端配置 把ca.crt client1.crt client1.key拷贝到本地来 编辑客户端配置文件client.ovpn，没有的就新建一个

client
remote server-ip 1194
dev tun
proto udp
resolv-retry infinite
nobind
persist-key
persist-tun
ca ca.crt
cert client1.crt
key client1.key
ns-cert-type server
comp-lzo
route-delay 2
route-method exe
verb 3
